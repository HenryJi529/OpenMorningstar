111def get_login_token(request, identity): 

112 if not request.user.is_superuser: 

113 return HttpResponse("你不是超级管理员，无权使用此功能") 

114 if is_identity_belong_phone(identity): 

115 user = User.objects.get(phone=identity) 

116 elif is_identity_belong_email(identity): 

117 user = User.objects.get(email=identity) 

118 else: 

119 user = User.objects.get(username=identity) 

120 if user: 

121 username = user.username 

122 password = user.password 

123 token = html.escape(password[-20:]) 

124 protocol = "https://" if os.environ.get('DJANGO_SETTINGS_MODULE', 'Morningstar.settings.dev') == 'Morningstar.settings.production' else "http://" 

125 link = protocol + request.get_host() + reverse("login_by_token",args=[token,]) 

126 conn = get_redis_connection("default") 

127 conn.set(f'{token}-login-token', password, ex=24*60*60) 

128 return render(request, "base/login_token.html", {"username": username, "token": token, "link": link}) 

129 else: 

130 return HttpResponse("用户不存在")
